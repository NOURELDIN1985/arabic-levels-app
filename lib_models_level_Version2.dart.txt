class Level {
  final int id;
  final String letter;
  final String vowel;
  final String display;
  final String image;
  final String audioLetter;
  final String audioWord;
  final List<String> choices;
  final int correctChoiceIndex;

  Level({
    required this.id,
    required this.letter,
    required this.vowel,
    required this.display,
    required this.image,
    required this.audioLetter,
    required this.audioWord,
    required this.choices,
    required this.correctChoiceIndex,
  });

  factory Level.fromJson(Map<String, dynamic> json) {
    return Level(
      id: json['id'],
      letter: json['letter'] ?? '',
      vowel: json['vowel'] ?? '',
      display: json['display'] ?? '',
      image: json['image'] ?? '',
      audioLetter: json['audioLetter'] ?? '',
      audioWord: json['audioWord'] ?? '',
      choices: List<String>.from(json['choices'] ?? []),
      correctChoiceIndex: json['correctChoiceIndex'] ?? 0,
    );
  }
}
